def create
  if @user.authenticate_by_phone(create_params.dig(:otp))
    start_new_session_for @user
    token = JwtAuthenticate.encode(user_id: @user.id)
    ok(token: token)
  else
    bad_request(Otp::Errors::InvalidOtpError.new.message)
  end
end
